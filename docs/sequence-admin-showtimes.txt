@startuml
actor Client as User

boundary "ManageShowtimesListScreen" as ManageShowtimesList
boundary "ManageShowtimeFormScreen" as ManageShowtimeForm

control "ShowtimeController" as ShowtimeController
control "ShowtimeService" as ShowtimeService

participant "ShowtimeRepository" as ShowtimeRepository
participant "ShowtimeSeatRepository" as ShowtimeSeatRepository
participant "SeatRepository" as SeatRepository

entity "Showtime" as Showtime
entity "ShowtimeSeat" as ShowtimeSeat
entity "Seat" as Seat

== Xem danh sách xuất chiếu ==
User -> ManageShowtimesList : Mở danh sách xuất chiếu
ManageShowtimesList -> ShowtimeController : GET /api/showtimes?movieId=...&cinemaId=...&date=...
ShowtimeController -> ShowtimeService : getShowtimesBy... (theo filter)
ShowtimeService -> ShowtimeRepository : findBy... (theo filter)
ShowtimeRepository -> Showtime : Truy vấn Showtime
Showtime --> ShowtimeRepository : Danh sách Showtime
ShowtimeRepository --> ShowtimeService : Danh sách Showtime
ShowtimeService --> ShowtimeController : Danh sách Showtime
ShowtimeController --> ManageShowtimesList : Danh sách Showtime
ManageShowtimesList --> User : Hiển thị danh sách

== Xem chi tiết xuất chiếu ==
User -> ManageShowtimesList : Chọn 1 xuất chiếu
ManageShowtimesList -> ShowtimeController : GET /api/showtimes/{id}
ShowtimeController -> ShowtimeService : getShowtimeById(id)
ShowtimeService -> ShowtimeRepository : findById(id)
ShowtimeRepository -> Showtime : Truy vấn Showtime
Showtime --> ShowtimeRepository : Showtime
ShowtimeRepository --> ShowtimeService : Showtime
ShowtimeService --> ShowtimeController : Showtime
ShowtimeController --> ManageShowtimesList : Showtime
ManageShowtimesList --> User : Hiển thị chi tiết

== Thêm xuất chiếu ==
User -> ManageShowtimeForm : Mở form thêm xuất chiếu
ManageShowtimeForm -> ShowtimeController : POST /api/showtimes
ShowtimeController -> ShowtimeService : createShowtime(showtime)
ShowtimeService -> ShowtimeRepository : save(showtime)
ShowtimeRepository -> Showtime : Lưu Showtime
Showtime --> ShowtimeRepository : Showtime đã lưu
ShowtimeRepository --> ShowtimeService : Showtime đã lưu

ShowtimeService -> SeatRepository : findByRoomId(roomId)
SeatRepository -> Seat : Truy vấn Seat
Seat --> SeatRepository : Danh sách Seat
SeatRepository --> ShowtimeService : Danh sách Seat

ShowtimeService -> ShowtimeSeatRepository : saveAll(showtimeSeats)
ShowtimeSeatRepository -> ShowtimeSeat : Lưu ShowtimeSeat
ShowtimeSeat --> ShowtimeSeatRepository : Đã lưu
ShowtimeSeatRepository --> ShowtimeService : OK

ShowtimeService --> ShowtimeController : Showtime
ShowtimeController --> ManageShowtimeForm : Showtime
ManageShowtimeForm --> User : Thêm thành công

== Sửa xuất chiếu ==
User -> ManageShowtimeForm : Mở form sửa xuất chiếu
ManageShowtimeForm -> ShowtimeController : PUT /api/showtimes/{id}
ShowtimeController -> ShowtimeService : updateShowtime(id, showtime)
ShowtimeService -> ShowtimeRepository : findById(id)
ShowtimeRepository -> Showtime : Truy vấn Showtime
Showtime --> ShowtimeRepository : Showtime
ShowtimeRepository --> ShowtimeService : Showtime

ShowtimeService -> ShowtimeRepository : save(showtime)
ShowtimeRepository -> Showtime : Lưu Showtime
Showtime --> ShowtimeRepository : Showtime đã cập nhật
ShowtimeRepository --> ShowtimeService : Showtime đã cập nhật

ShowtimeService --> ShowtimeController : Showtime
ShowtimeController --> ManageShowtimeForm : Showtime
ManageShowtimeForm --> User : Cập nhật thành công

== Xóa xuất chiếu ==
User -> ManageShowtimesList : Chọn xóa xuất chiếu
ManageShowtimesList -> ShowtimeController : DELETE /api/showtimes/{id}
ShowtimeController -> ShowtimeService : deleteShowtime(id)
ShowtimeService -> ShowtimeRepository : findById(id)
ShowtimeRepository -> Showtime : Truy vấn Showtime
Showtime --> ShowtimeRepository : Showtime
ShowtimeRepository --> ShowtimeService : Showtime

ShowtimeService -> ShowtimeSeatRepository : deleteByShowtimeId(id)
ShowtimeSeatRepository -> ShowtimeSeat : Xóa ShowtimeSeat
ShowtimeSeat --> ShowtimeSeatRepository : Đã xóa
ShowtimeSeatRepository --> ShowtimeService : OK

ShowtimeService -> ShowtimeRepository : delete(showtime)
ShowtimeRepository -> Showtime : Xóa Showtime
Showtime --> ShowtimeRepository : Đã xóa
ShowtimeRepository --> ShowtimeService : OK

ShowtimeService --> ShowtimeController : OK
ShowtimeController --> ManageShowtimesList : 204 No Content
ManageShowtimesList --> User : Xóa thành công
@enduml
