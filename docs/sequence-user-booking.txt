@startuml
actor Customer as User

' Frontend (screens)
boundary "BookingScreen" as BookingScreen
boundary "PaymentScreen" as PaymentScreen

' Controller
control "SeatController" as SeatController
control "BookingController" as BookingController
control "PaymentController" as PaymentController

' Service
control "BookingService" as BookingService
control "PaymentService" as PaymentService

' Repository
participant "ShowtimeSeatRepository" as ShowtimeSeatRepo
participant "BookingRepository" as BookingRepo
participant "BookingSeatRepository" as BookingSeatRepo
participant "ComboRepository" as ComboRepo
participant "BookingComboRepository" as BookingComboRepo
participant "PaymentRepository" as PaymentRepo

' Models
entity "Booking" as Booking
entity "Payment" as Payment
entity "ShowtimeSeat" as ShowtimeSeat

' External Gateway
participant "VNPAY" as VNPAY

autonumber 1

== Chon ghe ==
User -> BookingScreen : Select showtime, seats
BookingScreen -> SeatController : GET /api/seats/showtime/{showtimeId}
SeatController -> ShowtimeSeatRepo : findByShowtimeId(showtimeId)
ShowtimeSeatRepo -> ShowtimeSeat : Truy van ghe
ShowtimeSeat --> ShowtimeSeatRepo : Danh sach ghe
ShowtimeSeatRepo --> SeatController : Seats
SeatController --> BookingScreen : Seats
BookingScreen --> User : Show seat map

== Tao booking (combo, tong tien) ==
User -> PaymentScreen : Choose combos, confirm
PaymentScreen -> BookingController : POST /api/bookings/create-simple
note right of PaymentScreen
  body: showtimeId, showtimeSeatIds,
  combos, totalAmount
end note
BookingController -> BookingService : createBooking(...)
BookingService -> ShowtimeSeatRepo : findAllByIdForUpdate(seatIds)
ShowtimeSeatRepo --> BookingService : Seats locked
BookingService -> ComboRepo : findAllById(comboIds)
ComboRepo --> BookingService : Combos
BookingService -> BookingRepo : save(Booking PENDING_PAYMENT)
BookingRepo -> Booking : Luu booking
Booking --> BookingRepo : bookingId
BookingRepo --> BookingService : bookingId
BookingService -> BookingSeatRepo : save(BookingSeat)
BookingService -> BookingComboRepo : save(BookingCombo)
BookingService --> BookingController : bookingId
BookingController --> PaymentScreen : 200 {id, status}

== Tao URL thanh toan ==
PaymentScreen -> PaymentController : GET /api/payment/pay?bookingId={id}
PaymentController -> PaymentService : createPaymentUrl(bookingId)
PaymentService -> BookingRepo : findById(bookingId)
BookingRepo -> Booking : Lay booking
Booking --> BookingRepo : Booking
BookingRepo --> PaymentService : Booking
PaymentService -> PaymentRepo : save(Payment INIT)
PaymentRepo -> Payment : Luu payment
Payment --> PaymentRepo : OK
PaymentRepo --> PaymentService : OK
PaymentService --> PaymentController : paymentUrl
PaymentController --> PaymentScreen : 202 {paymentUrl}
PaymentScreen -> VNPAY : Open paymentUrl

== VNPAY callback ==
VNPAY -> PaymentController : GET /api/payment/vnpIpn
PaymentController -> PaymentService : handleVnpIpn(params)
PaymentService -> PaymentRepo : update Payment
PaymentService -> BookingRepo : update Booking
PaymentService -> BookingSeatRepo : findByBookingId(bookingId)
PaymentService -> ShowtimeSeatRepo : saveAll(BOOKED)
PaymentService --> PaymentController : IPN ack

VNPAY -> PaymentController : GET /api/payment/vnpReturn
PaymentController -> PaymentService : handleVnpReturn(params)
PaymentController --> PaymentScreen : deep-link moviebooking://payment-result

== Xac nhan thanh toan ==
PaymentScreen -> PaymentController : GET /api/payment/status?bookingId={id}
PaymentController -> PaymentService : getPaymentStatus(bookingId)
PaymentService -> BookingRepo : findById(bookingId)
BookingRepo --> PaymentService : CONFIRMED
PaymentService --> PaymentController : SUCCESS
PaymentController --> PaymentScreen : SUCCESS
PaymentScreen --> User : Show payment success
@enduml
