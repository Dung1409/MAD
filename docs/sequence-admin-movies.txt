@startuml
actor Client as User

boundary "ManageMoviesListScreen" as ManageMoviesList
boundary "ManageMovieFormScreen" as ManageMovieForm

control "AdminMovieController" as AdminMovieController
control "MovieService" as MovieService

participant "MovieRepository" as MovieRepository
participant "GenreRepository" as GenreRepository

entity "Movie" as Movie
entity "Genre" as Genre

== Xem danh sách phim ==
User -> ManageMoviesList : Mở màn hình danh sách
ManageMoviesList -> AdminMovieController : GET /api/admin/movies/status/{status} (tuỳ chọn)
AdminMovieController -> MovieService : getMoviesByStatus(status)
MovieService -> MovieRepository : findByStatus(status)
MovieRepository -> Movie : Truy vấn Movie
Movie --> MovieRepository : Danh sách Movie
MovieRepository --> MovieService : Danh sách Movie
MovieService --> AdminMovieController : MoviesListResponse
AdminMovieController --> ManageMoviesList : MoviesListResponse
ManageMoviesList --> User : Hiển thị danh sách

== Thêm phim ==
User -> ManageMovieForm : Mở form thêm phim
ManageMovieForm -> AdminMovieController : POST /api/admin/movies
AdminMovieController -> MovieService : createMovie(request)
MovieService -> GenreRepository : findById(genreId)*
GenreRepository -> Genre : Truy vấn Genre
Genre --> GenreRepository : Genre
GenreRepository --> MovieService : Genre
MovieService -> MovieRepository : save(movie)
MovieRepository -> Movie : Lưu Movie
Movie --> MovieRepository : Movie đã lưu
MovieRepository --> MovieService : Movie
MovieService --> AdminMovieController : MovieResponse
AdminMovieController --> ManageMovieForm : MovieResponse
ManageMovieForm --> User : Thông báo thêm thành công

== Sửa phim ==
User -> ManageMovieForm : Mở form sửa phim
ManageMovieForm -> AdminMovieController : PUT /api/admin/movies/{id}
AdminMovieController -> MovieService : updateMovie(id, request)
MovieService -> MovieRepository : findById(id)
MovieRepository -> Movie : Truy vấn Movie
Movie --> MovieRepository : Movie
MovieRepository --> MovieService : Movie
MovieService -> GenreRepository : findById(genreId)*
GenreRepository -> Genre : Truy vấn Genre
Genre --> GenreRepository : Genre
GenreRepository --> MovieService : Genre
MovieService -> MovieRepository : save(movie)
MovieRepository -> Movie : Lưu Movie
Movie --> MovieRepository : Movie đã cập nhật
MovieRepository --> MovieService : Movie
MovieService --> AdminMovieController : MovieResponse
AdminMovieController --> ManageMovieForm : MovieResponse
ManageMovieForm --> User : Thông báo cập nhật thành công

== Xóa phim ==
User -> ManageMoviesList : Chọn xóa phim
ManageMoviesList -> AdminMovieController : DELETE /api/admin/movies/{id}
AdminMovieController -> MovieService : deleteMovie(id)
MovieService -> MovieRepository : findById(id)
MovieRepository -> Movie : Truy vấn Movie
Movie --> MovieRepository : Movie
MovieRepository --> MovieService : Movie
MovieService -> MovieRepository : delete(movie)
MovieRepository -> Movie : Xóa Movie
Movie --> MovieRepository : Đã xóa
MovieRepository --> MovieService : OK
MovieService --> AdminMovieController : OK
AdminMovieController --> ManageMoviesList : 204 No Content
ManageMoviesList --> User : Xóa thành công
@enduml
